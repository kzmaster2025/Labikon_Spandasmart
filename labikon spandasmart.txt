        const message = `Halo ${teacherName}, saya ingin berkonsultasi dengan Bapak/Ibu mengenai masalah BK. Terima kasih.`;
        const encodedMessage = encodeURIComponent(message);

        window.open(whatsappUrl, '_blank');
        closeWhatsAppModal();
    }
